from subprocess import call

bam_files = listdir("bam_files/")
annotate_files = [x[:-3] + "gtf" for x in bam_files]

for (i, file) in enumerate(bam_files):
    cmd = "stringtie bam_files/%s -o annotation_files/%s -G Mus_musculus.GRCm38.98.19chr.gtf" % (file, annotate_files[i])
    call(cmd, shell=True)

import pandas as pd

names = [x[:-4] for x in new_bam_files]

corr = df.corr(method="spearman")

from sklearn.preprocessing import StandardScaler
from sklearn.decomposition import PCA

x = StandardScaler().fit_transform(df.T)

pca = PCA(n_components=2)

principalComponents = pca.fit_transform(x)
newdf = pd.DataFrame(data=principalComponents,
                    columns = ["pc1", "pc2"])
finaldf = pd.concat([newdf, pd.Series(df.columns)], axis=1)

import matplotlib.pyplot as plt

fig = plt.figure(figsize=(8,8))
ax = fig.add_subplot(1,1,1)
ax.set_xlabel("PC1")
ax.set_ylabel("PC2")
ax.set_title("a")

targets = list(df.columns)
col1 = "#ffff00"
col2 = "#ff0000"
colors = [col1, col2, col2, col2, col1, 
         col1, col1, col2, col2, col1]
for target, color in zip(targets,colors):
    indicesToKeep = finaldf[0] == target
    ax.scatter(finaldf.loc[indicesToKeep, "pc1"],
              finaldf.loc[indicesToKeep, "pc2"],
              c = color, s = 50)
ax.legend(targets)
ax.grid()
